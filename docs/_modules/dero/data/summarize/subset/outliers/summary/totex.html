    table = lt.Table.from_panel_name_df_dict(
        df_dict,
        include_index=True,
        top_left_corner_labels='Variable',
        caption=caption,
        landscape=True
    )

    if as_document:
        return Document(
            table,
            author=author,
            title=caption
        )

    return table
