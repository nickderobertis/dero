def _by_id_count_df(df: pd.DataFrame, id_col: str) -> pd.DataFrame:
    by_id_count_df = pd.DataFrame(
        data=[
            (len(df[id_col].dropna().unique()),)
        ],
        columns=['#']
    )

    return by_id_count_df
