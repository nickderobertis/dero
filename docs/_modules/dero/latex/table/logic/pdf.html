            three_part_table_kwargs=three_part_table_kwargs,
            **table_kwargs
        )


    document_to_pdf_and_move(
        table_or_document,
        outfolder=out_folder,
        outname=out_name,
        as_document=as_document,
        move_folder_name='Tables'
    )
